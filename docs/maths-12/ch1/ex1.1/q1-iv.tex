    \textbf{Find $f(x^2+4)$}

    \begin{enumerate}
        \item When $f(x) = x^2 - x$
        \begin{align*}
            f(x^2+4) &= (x^2+4)^2 - (x^2+4) \\
                &= \left[(x^2)^2 + 2(x^2)(4) + 4^2\right] - x^2 - 4 \\
                &= x^4 + 8x^2+ 16 - x^2 -4 \\
                &= x^4+7x^2+12
        \end{align*}

        \item When $f(x) = \sqrt{x + 4}$
        \begin{align*}
            f(x^2+4) &= \sqrt{(x^2+4) + 4}\\
                &= \sqrt{x^2+8}
        \end{align*}
    \end{enumerate}
