    \textbf{Find $f(-2)$}

    \begin{enumerate}
        \item When $f(x) = x^2 - x$
        \begin{align*}
            f(-2) &= (-2)^2 - (-2) \\
                &= 4+2 \\
                &= 6
        \end{align*}

        \item When $f(x) = \sqrt{x + 4}$
        \begin{align*}
            f(-2) &= \sqrt{-2 + 4} \\
                &= \sqrt{2}
        \end{align*}
    \end{enumerate}
