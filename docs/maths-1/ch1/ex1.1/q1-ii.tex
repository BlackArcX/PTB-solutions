    \textbf{Find $f(0)$}

    \begin{enumerate}
        \item When $f(x) = x^2 - x$
        \begin{align*}
            f(0) &= (0)^2 - (0) \\
                &= 0
        \end{align*}

        \item When $f(x) = \sqrt{x + 4}$
        \begin{align*}
            f(0) &= \sqrt{0 + 4} \\
                &= \sqrt{4}
        \end{align*}
    \end{enumerate}
