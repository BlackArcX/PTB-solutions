    \textbf{Find $\frac{f(a+h)-f(a)}{h}$ and simplify, where $f(x)=6x-9$}

    \begin{align*}
        f(a+h) &= 6(a+h) - 9 \\
            &= 6a + 6h -9 \\
        \\
        f(a) &= 6(a) - 9 \\
            &= 6a - 9 \\
        \\
        \frac{f(a+h)-f(a)}{h} &= \frac{6a + 6h - 9 - (6a -9)}{h} \\
            &= \frac{6h}{h} \\
            &= 6
    \end{align*}
