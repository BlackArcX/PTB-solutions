    \textbf{Find $f(x-1)$}

    \begin{enumerate}
        \item When $f(x) = x^2 - x$
        \begin{align*}
            f(x-1) &= (x-1)^2 - (x-1)\\
                &= x^2 - 2x + 1 - x + 1\\
                &= x^2 - 3x + 2
        \end{align*}

        \item When $f(x) = \sqrt{x + 4}$
        \begin{align*}
            f(x-1) &= \sqrt{(x-1) + 4}\\
                &=\sqrt{x + 3}
        \end{align*}
    \end{enumerate}
